int nivel = 2;
int estado = 0;

void setup() {
  pinMode(nivel, INPUT_PULLUP);
  Serial.begin(9600);
}

void loop() {
  estado = digitalRead(nivel);

  if (estado == LOW) {
    Serial.println("VACIO");
  } else {
    Serial.println("OK");
  }

  delay(300);
}
